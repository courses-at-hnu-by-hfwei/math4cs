\begin{frame}{}
  \begin{theorem}[Dov Jarden (1953)]
    \[
      \exists a, b \in \mathbb{R} \setminus \mathbb{Q}: a^{b} \in \mathbb{Q}.
    \]
  \end{theorem}

  \pause
  \[
      \sqrt{2} \in \mathbb{R} \setminus \mathbb{Q}
  \]

  \pause
  \[
      \sqrt{2}^{\sqrt{2}} \in \mathbb{Q} \quad \text{or} \quad \sqrt{2}^{\sqrt{2}} \in \mathbb{R} \setminus \mathbb{Q}
  \]

  \pause
  \[
      (\sqrt{2}^{\sqrt{2}})^{\sqrt{2}} \pause = \sqrt{2}^{\sqrt{2} \cdot \sqrt{2}} = \sqrt{2}^2 = 2 \in \mathbb{Q}
  \]

  \pause
  \centerline{\red{$Q:$} 这是构造性证明吗? \pause 这是反证法吗?}

  \pause

  \[
    a = \sqrt{2} \qquad b = 2 \log_{2} 3 \qquad (a^b = 3)
  \]
\end{frame}
